/**
 * Esta clase proporciona métodos para realizar solicitudes HTTP GET, PUT, POST y DELETE a través de OAuth2.
 * @module BigQueryOAuthManager
 * @author Nicolas Cuervo <jhoan.cuervo@servinformacion.com>
 */

export class BigQueryOAuthManager {

    /**
     * Constructor que inicializa un objeto con credenciales e información de tabla
     * para interactuar con una base de datos.
     * @constructor
     * @param {object} credentials - Un objeto que contiene detalles de autenticación para la base de datos.
     * @param {string} projectId - El ID del proyecto donde se encuentra la tabla.
     * @param {string} datasetId - El ID del conjunto de datos que contiene la tabla.
     * @param {string} tableName - El nombre de la tabla con la que interactuar.
     *
     * @throws {TypeError} - Si alguno de los parámetros requeridos falta o no es del tipo esperado.
    */
    constructor(credentials, projectId, datasetId, tableName) {
        if (typeof credentials !== 'object' || credentials === null) {
            throw new TypeError('Las credenciales deben ser un objeto');
        }
        if (typeof projectId !== 'string' || projectId.length === 0) {
            throw new TypeError('idProyecto debe ser una cadena no vacía');
        }
        if (typeof datasetId !== 'string' || datasetId.length === 0) {
            throw new TypeError('idConjuntoDatos debe ser una cadena no vacía');
        }
        if (typeof tableName !== 'string' || tableName.length === 0) {
            throw new TypeError('nombreTabla debe ser una cadena no vacía');
        }
        this.credentials = credentials;
        this.projectId = projectId;
        this.datasetId = datasetId;
        this.tableName = tableName;
        this.tableFullName = `${projectId}.${datasetId}.${tableFullName}`;
    }

    /**
     * Obtiene datos de la tabla BigQuery especificada.
     * @param {string} inputs (opcional) - Los campos que se desean recuperar de la tabla. Se puede usar '*' para todos los campos.
     * @param {string} condition (opcional) - La condición WHERE para filtrar los resultados.
     * @param {boolean} parse (opcional) - Indica si se deben parsear los resultados a un array de objetos (predeterminado: true).
     *
     * @returns {array<object>|object} Un array de objetos que contienen los datos de las filas
     *   (si se activa la opción `parse`), o el objeto sin procesar devuelto por la consulta BigQuery
     *   (si se desactiva la opción `parse`). Si no se encuentran resultados o hay un error,
     *   se devuelve un objeto vacío.
     */
    get(inputs = '*', condition = '', parse = true) {
        const query = `SELECT ${inputs} FROM ${this.tableFullName} ${condition}`;
        const data = this.executeQuery(query);
        if (!data.jobComplete) {
            return this.get(condition, inputs, parse);
        }
        return parse ? this.parseResults(data) : data;
    }

    /**
     * Inserta datos en la tabla BigQuery especificada.
     * @param {string} inputs - Una cadena que representa los nombres de las columnas de la tabla
     *   en las que se insertarán los datos (separados por comas).
     * @param {string|array} values - Los valores a insertar en las columnas especificadas en `inputs`
     *   en las que se insertarán los datos (separados por comas).
     * @returns {object} El objeto sin procesar devuelto por la consulta BigQuery, que
     *   contiene información sobre la inserción realizada.
     */
    post(inputs, values) {
        const query = `INSERT INTO ${this.table} (${inputs}) VALUES ${values}`;
        return this.executeQuery(query);
    };

    /**
     * Actualiza filas en la tabla BigQuery especificada.
     * @param {string} inputs - Una cadena que representa los pares clave-valor de las columnas
     *   que se desean actualizar, separadas por comas (e.g., 'column1 = value1, column2 = value2').
     * @param {string} condition (opcional) - La condición WHERE para filtrar las filas que se actualizarán.
     * @returns {object} El objeto sin procesar devuelto por la consulta BigQuery, que
     *   contiene información sobre la actualización realizada.
     */
    put(inputs, condition = '') {
        const query = `UPDATE ${this.tableFullName} SET ${inputs} ${condition}`;
        return this.executeQuery(query);
    }

    /**
     * Elimina filas de la tabla BigQuery especificada.
     * @param {string} condition (opcional) - La condición WHERE para filtrar las filas que se eliminarán.
     * @returns {object} El objeto sin procesar devuelto por la consulta BigQuery, que
     *   contiene información sobre la eliminación realizada.
     */
    del(condition = '') {
        const query = `DELETE FROM ${this.tableFullName} ${condition}`;
        return this.executeQuery(query);
    }

    /**
     * Ejecuta una consulta SQL en BigQuery.
     * @param {string} query - La consulta SQL a ejecutar.
     * @returns {object} Un objeto que contiene el resultado de la consulta en caso de éxito,
     *   o un objeto con la propiedad `error` que contiene el mensaje de error en caso de fallo.
     */
    executeQuery(query) {
        try {
            const bigQueryService = this.getBQService();
            bigQueryService.reset();
            if (!bigQueryService.hasAccess()) {
                throw new Error('Access to BigQuery service is not available.');
            }
            const url = `https://bigquery.googleapis.com/bigquery/v2/projects/${this.projectId}/queries`;
            const request = { query: query, useLegacySql: false };
            const options = {
                method: 'post',
                headers: {
                    Authorization: `Bearer ${bigQueryService.getAccessToken()}`,
                    'Content-Type': 'application/json'
                },
                payload: JSON.stringify(request),
                muteHttpExceptions: true
            };
            const response = UrlFetchApp.fetch(url, options);
            return JSON.parse(response.getContentText());
        } catch (error) {
            return { error: error.message };
        }
    }
    
    /**
     * Obtiene el servicio OAuth2 para interactuar con BigQuery.
     * @returns {object} Un objeto de servicio OAuth2 configurado para BigQuery.
     */
    getBQService() {
        return OAuth2.createService('BigQuery')
            .setTokenUrl('https://accounts.google.com/o/oauth2/token')
            .setPrivateKey(this.credentials.private_key)
            .setIssuer(this.credentials.client_email)
            .setPropertyStore(PropertiesService.getScriptProperties())
            .setLock(LockService.getUserLock())
            .setScope([
                'https://www.googleapis.com/auth/cloud-platform',
                'https://www.googleapis.com/auth/drive',
                'https://www.googleapis.com/auth/bigquery.readonly',
                'https://www.googleapis.com/auth/bigquery.insertdata',
                'https://www.googleapis.com/auth/bigquery'
            ]);
    }

    /**
     * Parsea los resultados de una consulta BigQuery y los convierte en un array de objetos.
     * @param {object} results - El objeto que contiene los resultados de la consulta BigQuery.
     *   Se espera que el objeto tenga las propiedades `totalRows`, `schema`, y `rows`.
     *   - `totalRows`: El número total de filas devueltas por la consulta.
     *   - `schema`: El esquema de la tabla con información sobre los campos.
     *   - `rows`: Un array de objetos que representan las filas de datos.
     * @returns {array<object>} Un array de objetos que contienen los datos de las filas,
     *   o un array vacío si no hay resultados o el formato es inválido.
     */
    parseResults(results) {
        if (results.totalRows > 0 && results.schema && results.schema.fields) {
            const names = results.schema.fields.map(field => field.name);
            return results.rows.map(row => {
                const obj = {};
                row.f.forEach((field, index) => {
                    obj[names[index]] = field.v !== undefined ? field.v : '';
                });
                return obj;
            });
        } else {
            return [];
        }
    }
}
